```r
read_insula_data <- function(file_path) {
    #' 读取脑岛投射数据
    #' @param file_path Excel文件路径
    #' @return 包含所有数据的列表
    
    cat("📖 读取数据:", file_path, "\n")
    
    # 读取各sheet
    df_projection <- read_excel(file_path, sheet = "Projection_Data")
    df_metadata <- read_excel(file_path, sheet = "Metadata")
    df_regions <- read_excel(file_path, sheet = "Region_Info")
    
    # 设置行名
    df_projection <- df_projection %>% 
        column_to_rownames(var = colnames(df_projection)[1])
    
    # 识别脑岛相关列
    insula_cols <- df_regions %>% 
        filter(Is_Insula == TRUE) %>% 
        pull(Region_Name)
    
    cat("  ✓ 投射矩阵:", nrow(df_projection), "x", ncol(df_projection), "\n")
    cat("  ✓ 元数据:", nrow(df_metadata), "行\n")
    cat("  ✓ 脑岛区域:", length(insula_cols), "个\n")
    
    return(list(
        projection = df_projection,
        metadata = df_metadata,
        regions = df_regions,
        insula_cols = insula_cols
    ))
}
```

```r
get_insula_soma_neurons <- function(data_list) {
    #' 获取胞体在脑岛的神经元
    
    insula_neurons <- data_list$metadata %>%
        filter(Soma_Region == "Insula")
    
    # 提取这些神经元的投射数据
    insula_projection <- data_list$projection[insula_neurons$Neuron_ID, ]
    
    cat("✓ 胞体在脑岛的神经元:", nrow(insula_neurons), "个\n")
    cat("  - FI:", sum(insula_neurons$Soma_Subregion == "FI"), "\n")
    cat("  - Iai:", sum(insula_neurons$Soma_Subregion == "Iai"), "\n")
    cat("  - Ig:", sum(insula_neurons$Soma_Subregion == "Ig"), "\n")
    
    return(list(
        metadata = insula_neurons,
        projection = insula_projection
    ))
}
```

```r
plot_insula_soma_heatmap <- function(insula_data, data_list, output_prefix) {
    #' 绘制胞体在脑岛的神经元投射热图
    
    mat <- as.matrix(insula_data$projection)
    meta <- insula_data$metadata
    
    # 确保顺序一致
    meta <- meta[match(rownames(mat), meta$Neuron_ID), ]
    
    # 创建顶部注释
    ha_top <- HeatmapAnnotation(
        # 脑岛亚区
        Insula_Subregion = meta$Soma_Subregion,
        # 半球
        Hemisphere = meta$Soma_Hemisphere,
        # 神经元类型
        Neuron_Type = meta$Neuron_Type,
        # 总轴突长度（条形图）
        Axon_Length = anno_barplot(
            log10(meta$Total_Axon_Length),
            gp = gpar(fill = "steelblue", col = NA),
            height = unit(1.2, "cm")
        ),
        
        col = list(
            Insula_Subregion = insula_subregion_colors,
            Hemisphere = hemisphere_colors,
            Neuron_Type = c("IT" = "#4DAF4A", "PT" = "#E41A1C", 
                           "CT" = "#FF7F00", "ITc" = "#984EA3")
        ),
        
        annotation_name_side = "left",
        annotation_name_gp = gpar(fontsize = 9, fontface = "bold"),
        show_legend = c(TRUE, TRUE, TRUE, FALSE)
    )
    
    # 识别脑岛列和非脑岛列
    insula_cols <- data_list$insula_cols
    non_insula_cols <- setdiff(colnames(mat), insula_cols)
    
    # 重新排列列：脑岛列在前，非脑岛列在后
    col_order <- c(intersect(insula_cols, colnames(mat)), non_insula_cols)
    mat <- mat[, col_order]
    
    # 创建列注释（标记脑岛列）
    is_insula_col <- colnames(mat) %in% insula_cols
    col_anno <- HeatmapAnnotation(
        Is_Insula = ifelse(is_insula_col, "Insula", "Other"),
        col = list(Is_Insula = c("Insula" = "#E41A1C", "Other" = "gray80")),
        show_legend = TRUE,
        height = unit(0.3, "cm")
    )
    
    # 创建热图
    ht <- Heatmap(
        mat,
        name = "Log Axon Length",
        col = projection_col_fun(mat),
        
        # 聚类
        cluster_rows = TRUE,
        cluster_columns = TRUE,
        row_dend_gp = gpar(col = "gray40", lwd = 0.6),
        column_dend_gp = gpar(col = "gray40", lwd = 0.6),
        
        # 标签
        row_names_side = "left",
        row_names_gp = gpar(fontsize = 7),
        column_names_side = "bottom",
        column_names_gp = gpar(fontsize = 7),
        column_names_rot = 45,
        
        # 注释
        top_annotation = ha_top,
        bottom_annotation = col_anno,
        
        # 尺寸
        width = unit(18, "cm"),
        height = unit(max(10, nrow(mat) * 0.4), "cm"),
        
        # 标题
        column_title = "Target Brain Regions",
        column_title_gp = gpar(fontsize = 11, fontface = "bold"),
        row_title = "Insula Neurons",
        row_title_gp = gpar(fontsize = 11, fontface = "bold")
    )
    
    # 保存
    pdf(paste0(output_prefix, "_InsulaSoma_Heatmap.pdf"), 
        width = 16, height = max(10, nrow(mat) * 0.15))
    draw(ht, merge_legend = TRUE)
    dev.off()
    
    cat("✅ 脑岛神经元热图已保存\n")
    return(ht)
}
```

```r
plot_cytoarchitecture_heatmap <- function(insula_data, output_prefix) {
    #' 按细胞构筑（Ig/Id/Ia）分组的热图
    
    mat <- as.matrix(insula_data$projection)
    meta <- insula_data$metadata
    
    # 添加细胞构筑信息
    meta$Cytoarchitecture <- case_when(
        meta$Soma_Subregion %in% c("Ig") ~ "Granular",
        meta$Soma_Subregion %in% c("Id") ~ "Dysgranular",
        TRUE ~ "Agranular"
    )
    
    # 按细胞构筑排序
    meta <- meta %>% arrange(Cytoarchitecture, Soma_Subregion)
    mat <- mat[meta$Neuron_ID, ]
    
    # 创建行分割
    row_split <- meta$Cytoarchitecture
    
    # 创建热图
    ht <- Heatmap(
        mat,
        name = "Projection",
        col = projection_col_fun(mat),
        
        # 按细胞构筑分割
        row_split = row_split,
        row_gap = unit(3, "mm"),
        
        # 聚类（在每个组内聚类）
        cluster_row_slices = TRUE,
        cluster_columns = TRUE,
        
        # 标签
        row_names_gp = gpar(fontsize = 7),
        column_names_gp = gpar(fontsize = 7),
        column_names_rot = 45,
        
        # 标题
        column_title = "Target Regions",
        row_title = "Cytoarchitecture",
        
        width = unit(16, "cm"),
        height = unit(12, "cm")
    )
    
    pdf(paste0(output_prefix, "_Cytoarchitecture_Heatmap.pdf"), 
        width = 14, height = 10)
    draw(ht)
    dev.off()
    
    cat("✅ 细胞构筑分组热图已保存\n")
    return(ht)
}
```

```r
get_neurons_projecting_to_insula <- function(data_list, threshold = 0) {
    #' 获取投射到脑岛的神经元
    #' @param threshold 投射强度阈值
    
    insula_cols <- data_list$insula_cols
    mat <- data_list$projection
    
    # 筛选有投射到任何脑岛区域的神经元
    has_projection <- rowSums(mat[, insula_cols, drop = FALSE] > threshold) > 0
    
    projecting_neurons <- data_list$metadata[has_projection, ]
    projecting_mat <- mat[has_projection, ]
    
    cat("✓ 投射到脑岛的神经元:", nrow(projecting_neurons), "个\n")
    
    return(list(
        metadata = projecting_neurons,
        projection = projecting_mat,
        insula_projection = projecting_mat[, insula_cols, drop = FALSE]
    ))
}
```

### 3.3 投射模式聚类分析

```r
analyze_projection_patterns <- function(projecting_data, output_prefix) {
    #' 分析投射到脑岛的模式聚类
    
    mat <- projecting_data$insula_projection
    
    # 层次聚类
    dist_mat <- dist(mat, method = "euclidean")
    hc <- hclust(dist_mat, method = "ward.D2")
    
    # 切割聚类树（根据肘部法则或先验知识）
    k <- 5  # 可以调整
    clusters <- cutree(hc, k = k)
    
    # 添加聚类信息到metadata
    meta <- projecting_data$metadata
    meta$Projection_Cluster <- clusters[meta$Neuron_ID]
    
    cat("✓ 识别到", k, "个投射模式亚型\n")
    
    # 绘制带聚类标签的热图
    ht <- Heatmap(
        as.matrix(mat),
        name = "Projection",
        col = projection_col_fun(as.matrix(mat)),
        
        cluster_rows = hc,
        row_split = k,
        row_gap = unit(2, "mm"),
        
        cluster_columns = TRUE,
        
        row_names_gp = gpar(fontsize = 6),
        column_names_gp = gpar(fontsize = 8),
        column_names_rot = 45,
        
        column_title = "Insula Subregions",
        row_title = "Projection Pattern Clusters",
        
        width = unit(12, "cm"),
        height = unit(14, "cm")
    )
    
    pdf(paste0(output_prefix, "_ProjectionPattern_Clusters.pdf"), 
        width = 12, height = 12)
    draw(ht)
    dev.off()
    
    # 保存聚类结果
    write.csv(meta, paste0(output_prefix, "_ProjectionClusters.csv"), 
              row.names = FALSE)
    
    cat("✅ 投射模式聚类分析完成\n")
    return(list(hc = hc, clusters = clusters, metadata = meta))
}
```

```r
plot_bilateral_comparison <- function(data_list, output_prefix) {
    #' 绘制左右脑岛投射对比热图
    
    # 识别左右脑岛列
    left_insula_cols <- grep("^Left_I", data_list$insula_cols, value = TRUE)
    right_insula_cols <- grep("^Right_I", data_list$insula_cols, value = TRUE)
    
    # 获取胞体在脑岛的神经元
    insula_neurons <- data_list$metadata %>%
        filter(Soma_Region == "Insula")
    
    mat <- data_list$projection[insula_neurons$Neuron_ID, ]
    
    # 提取左右投射
    left_proj <- mat[, left_insula_cols, drop = FALSE]
    right_proj <- mat[, right_insula_cols, drop = FALSE]
    
    # 确保列名匹配（去掉Left_/Right_前缀）
    colnames(left_proj) <- gsub("^Left_", "", colnames(left_proj))
    colnames(right_proj) <- gsub("^Right_", "", colnames(right_proj))
    
    # 只保留共同存在的列
    common_regions <- intersect(colnames(left_proj), colnames(right_proj))
    left_proj <- left_proj[, common_regions, drop = FALSE]
    right_proj <- right_proj[, common_regions, drop = FALSE]
    
    # 创建左右对比热图（并排）
    ht_left <- Heatmap(
        as.matrix(left_proj),
        name = "Left Insula",
        col = projection_col_fun(as.matrix(left_proj)),
        cluster_rows = TRUE,
        cluster_columns = TRUE,
        row_names_gp = gpar(fontsize = 6),
        column_names_gp = gpar(fontsize = 8),
        column_title = "Left Hemisphere Projection",
        width = unit(8, "cm")
    )
    
    ht_right <- Heatmap(
        as.matrix(right_proj),
        name = "Right Insula",
        col = projection_col_fun(as.matrix(right_proj)),
        cluster_rows = TRUE,
        cluster_columns = TRUE,
        row_names_gp = gpar(fontsize = 6),
        column_names_gp = gpar(fontsize = 8),
        column_title = "Right Hemisphere Projection",
        width = unit(8, "cm")
    )
    
    # 并排绘制
    pdf(paste0(output_prefix, "_Bilateral_Comparison.pdf"), 
        width = 18, height = 12)
    draw(ht_left + ht_right)
    dev.off()
    
    cat("✅ 双侧对比热图已保存\n")
    return(list(left = left_proj, right = right_proj))
}
```

### 4.2 不对称性指数热图

```r
plot_asymmetry_heatmap <- function(data_list, output_prefix) {
    #' 绘制不对称性指数热图
    
    # 获取左右投射
    bilateral <- plot_bilateral_comparison(data_list, output_prefix)
    
    # 计算不对称性指数
    # AI = (Left - Right) / (Left + Right + epsilon)
    epsilon <- 0.001  # 避免除零
    asymmetry <- (bilateral$left - bilateral$right) / 
                 (bilateral$left + bilateral$right + epsilon)
    
    # 创建红-蓝颜色方案（红：左>右，蓝：右>左）
    max_asym <- max(abs(asymmetry), na.rm = TRUE)
    col_fun <- colorRamp2(
        c(-max_asym, 0, max_asym),
        c("#2166AC", "#FFFFFF", "#B2182B")  # 蓝-白-红
    )
    
    # 创建热图
    ht <- Heatmap(
        as.matrix(asymmetry),
        name = "Asymmetry Index",
        col = col_fun,
        
        cluster_rows = TRUE,
        cluster_columns = TRUE,
        
        row_names_gp = gpar(fontsize = 7),
        column_names_gp = gpar(fontsize = 8),
        column_names_rot = 45,
        
        column_title = "Insula Subregions",
        row_title = "Neurons",
        
        width = unit(10, "cm"),
        height = unit(12, "cm")
    )
    
    pdf(paste0(output_prefix, "_Asymmetry_Index.pdf"), width = 12, height = 12)
    draw(ht)
    dev.off()
    
    cat("✅ 不对称性指数热图已保存\n")
    
    # 统计显著性分析
    cat("\n📊 不对称性统计:\n")
    cat("  平均不对称性:", mean(asymmetry, na.rm = TRUE), "\n")
    cat("  标准差:", sd(asymmetry, na.rm = TRUE), "\n")
    
    return(asymmetry)
}
```
