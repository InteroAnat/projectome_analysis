## 1. ComplexHeatmap基础

### 1.1 最简单的热图

```r
library(ComplexHeatmap)
library(circlize)

# 基础热图
Heatmap(matrix, name = "legend_title")
```

### 1.2 常用参数

```r
Heatmap(
    matrix,
    name = "Legend Name",           # 图例标题
    col = color_palette,            # 颜色方案
    
    # 聚类
    cluster_rows = TRUE,            # 行聚类
    cluster_columns = TRUE,         # 列聚类
    
    # 标签
    row_names_side = "left",        # 行名位置
    column_names_side = "bottom",   # 列名位置
    row_names_gp = gpar(fontsize = 8),    # 行名字体
    column_names_gp = gpar(fontsize = 8), # 列名字体
    column_names_rot = 45,          # 列名旋转角度
    
    # 尺寸
    width = unit(10, "cm"),
    height = unit(8, "cm"),
    
    # 标题
    column_title = "Column Title",
    row_title = "Row Title"
)
```

### 1.3 颜色方案

```r
# 方法1: 使用colorRamp2（推荐）
col_fun <- colorRamp2(
    c(0, 2, 4, 6, 8),                                    # 数值断点
    c("#FFFFFF", "#C6DBEF", "#6BAED6", "#2171B5", "#08306B")  # 对应颜色
)

# 方法2: 使用内置颜色
Heatmap(matrix, col = viridis(100))
Heatmap(matrix, col = brewer.pal(9, "Blues"))

# 方法3: 自定义颜色向量
colors <- colorRampPalette(c("white", "blue", "navy"))(100)
```

## 2. 多维度注释

### 2.1 顶部注释 (HeatmapAnnotation)

```r
# 创建顶部注释
ha_top <- HeatmapAnnotation(
    # 分类变量
    Type = metadata$Neuron_Type,
    Cluster = metadata$Morph_Cluster,
    
    # 连续变量（条形图）
    Total_Length = anno_barplot(
        metadata$Total_Length,
        gp = gpar(fill = "steelblue"),
        height = unit(1.5, "cm")
    ),
    
    # 颜色映射
    col = list(
        Type = c("ITs" = "#4DAF4A", "ITi" = "#984EA3"),
        Cluster = c("1" = "red", "2" = "blue")
    ),
    
    annotation_name_side = "left"
)

# 应用注释
Heatmap(matrix, top_annotation = ha_top)
```

### 2.2 右侧注释 (rowAnnotation)

```r
# 创建右侧注释
ha_right <- rowAnnotation(
    Type = metadata$Neuron_Type,
    Length = anno_barplot(metadata$Total_Length),
    col = list(Type = type_colors)
)

# 应用注释
Heatmap(matrix, right_annotation = ha_right)
```

### 2.3 左侧注释

```r
ha_left <- rowAnnotation(
    foo = anno_block(
        gp = gpar(fill = c("red", "blue")),
        labels = c("Group1", "Group2")
    )
)
```

## 3. 热图分割与分组

### 3.1 按行分割

```r
# 按分类变量分割
Heatmap(matrix, row_split = metadata$Neuron_Type)

# 按聚类结果分割（k个组）
Heatmap(matrix, row_split = 5)

# 按多个变量分割
Heatmap(matrix, row_split = paste(metadata$Type, metadata$Cluster))
```

### 3.2 设置分割间隙

```r
Heatmap(
    matrix,
    row_split = metadata$Neuron_Type,
    row_gap = unit(3, "mm"),      # 行间间隙
    row_title = "Type",            # 分割标题
    row_title_gp = gpar(fontsize = 10)
)
```

### 3.3 按列分割

```r
Heatmap(matrix, column_split = region_groups)
```

## 4. 多个热图拼接

### 4.1 水平拼接 (+)

```r
ht1 <- Heatmap(matrix1, name = "Data1")
ht2 <- Heatmap(matrix2, name = "Data2")

# 水平拼接
ht1 + ht2

# 绘制
draw(ht1 + ht2)
```

### 4.2 垂直拼接 (%v%)

```r
# 垂直拼接
ht1 %v% ht2
```

### 4.3 复杂布局

```r
# 2x2布局
(ht1 + ht2) %v% (ht3 + ht4)
```

## 5. 针对251637数据的实战

### 5.1 完整的多注释热图

```r
# 准备数据
proj_matrix <- df_proj_length %>%
    select(-NeuronID, -Neuron_Type) %>%
    as.matrix()
rownames(proj_matrix) <- df_proj_length$NeuronID

# 筛选有投射的脑区
col_sums <- colSums(proj_matrix)
proj_matrix <- proj_matrix[, col_sums > 0]
proj_matrix_log <- log1p(proj_matrix)

# 合并聚类信息
df_merged <- df_summary %>%
    left_join(df_clustered %>% select(NeuronID, Morph_Cluster), by = "NeuronID")

# 确保顺序一致
df_merged <- df_merged[match(rownames(proj_matrix_log), df_merged$NeuronID), ]

# 定义颜色
type_colors <- c("ITs" = "#4DAF4A", "ITi" = "#984EA3", 
                 "ITc" = "#377EB8", "CT" = "#FF7F00")

cluster_colors <- c(
    "1" = "#E41A1C", "2" = "#377EB8", "3" = "#4DAF4A",
    "4" = "#984EA3", "5" = "#FF7F00", "6" = "#FFFF33",
    "7" = "#A65628", "8" = "#F781BF", "9" = "#999999"
)

# 创建顶部注释（多维度）
ha_top <- HeatmapAnnotation(
    # 分类注释
    Cluster = as.character(df_merged$Morph_Cluster),
    Type = df_merged$Neuron_Type,
    
    # 连续变量（条形图）
    Total_Length = anno_barplot(
        df_merged$Total_Length,
        gp = gpar(fill = "steelblue", col = NA),
        height = unit(1.2, "cm"),
        axis_param = list(gp = gpar(fontsize = 6))
    ),
    
    Terminal_Count = anno_barplot(
        df_merged$Terminal_Count,
        gp = gpar(fill = "orange", col = NA),
        height = unit(1.2, "cm"),
        axis_param = list(gp = gpar(fontsize = 6))
    ),
    
    # 颜色映射
    col = list(
        Cluster = cluster_colors,
        Type = type_colors
    ),
    
    annotation_name_side = "left",
    annotation_name_gp = gpar(fontsize = 9, fontface = "bold"),
    show_legend = c(TRUE, TRUE, FALSE, FALSE)
)

# 创建热图
col_fun <- colorRamp2(
    c(0, 2, 4, 6, 8),
    c("#FFFFFF", "#C6DBEF", "#6BAED6", "#2171B5", "#08306B")
)

ht <- Heatmap(
    proj_matrix_log,
    name = "Log Length",
    col = col_fun,
    
    # 聚类
    cluster_rows = TRUE,
    cluster_columns = TRUE,
    row_dend_gp = gpar(col = "gray40", lwd = 0.5),
    column_dend_gp = gpar(col = "gray40", lwd = 0.5),
    
    # 按聚类分割
    row_split = df_merged$Morph_Cluster,
    row_gap = unit(2, "mm"),
    
    # 标签
    row_names_side = "left",
    row_names_gp = gpar(fontsize = 4),
    column_names_side = "bottom",
    column_names_gp = gpar(fontsize = 5),
    column_names_rot = 45,
    show_row_names = FALSE,
    
    # 注释
    top_annotation = ha_top,
    
    # 标题
    column_title = "Target Brain Regions",
    column_title_gp = gpar(fontsize = 12, fontweight = "bold"),
    row_title = "Morphology Clusters (k=9)",
    row_title_gp = gpar(fontsize = 12, fontweight = "bold"),
    
    # 尺寸
    width = unit(16, "cm"),
    height = unit(18, "cm")
)

# 保存
pdf("advanced_heatmap_k9.pdf", width = 18, height = 20)
draw(ht, merge_legend = TRUE, heatmap_legend_side = "right")
dev.off()
```

### 5.2 左右对比热图

```r
# 识别左右脑区列
left_cols <- grep("^CL_", colnames(proj_matrix), value = TRUE)
right_cols <- grep("^CR_", colnames(proj_matrix), value = TRUE)

# 提取左右投射
left_proj <- proj_matrix[, left_cols, drop = FALSE]
right_proj <- proj_matrix[, right_cols, drop = FALSE]

# 标准化列名
colnames(left_proj) <- gsub("^CL_", "", colnames(left_proj))
colnames(right_proj) <- gsub("^CR_", "", colnames(right_proj))

# 共同区域
common_regions <- intersect(colnames(left_proj), colnames(right_proj))
left_proj <- left_proj[, common_regions, drop = FALSE]
right_proj <- right_proj[, common_regions, drop = FALSE]

# log转换
left_log <- log1p(left_proj)
right_log <- log1p(right_proj)

# 创建热图
col_fun <- colorRamp2(c(0, 2, 4, 6, 8), 
                      c("#FFFFFF", "#C6DBEF", "#6BAED6", "#2171B5", "#08306B"))

ht_left <- Heatmap(
    left_log,
    name = "Left",
    col = col_fun,
    cluster_rows = TRUE,
    cluster_columns = TRUE,
    row_names_gp = gpar(fontsize = 5),
    column_names_gp = gpar(fontsize = 7),
    column_title = "Left Hemisphere (CL)",
    width = unit(8, "cm")
)

ht_right <- Heatmap(
    right_log,
    name = "Right",
    col = col_fun,
    cluster_rows = TRUE,
    cluster_columns = TRUE,
    row_names_gp = gpar(fontsize = 5),
    column_names_gp = gpar(fontsize = 7),
    column_title = "Right Hemisphere (CR)",
    width = unit(8, "cm")
)

# 并排绘制
pdf("bilateral_comparison.pdf", width = 18, height = 14)
draw(ht_left + ht_right)
dev.off()
```

### 5.3 投射强度vs长度对比

```r
# 准备长度和强度矩阵
length_matrix <- df_proj_length %>%
    select(-NeuronID, -Neuron_Type) %>%
    as.matrix()
strength_matrix <- df_proj_strength %>%
    select(-NeuronID, -Neuron_Type) %>%
    as.matrix()

rownames(length_matrix) <- df_proj_length$NeuronID
rownames(strength_matrix) <- df_proj_strength$NeuronID

# 筛选相同区域
common_cols <- intersect(colnames(length_matrix), colnames(strength_matrix))
length_matrix <- length_matrix[, common_cols]
strength_matrix <- strength_matrix[, common_cols]

# log转换
length_log <- log1p(length_matrix)
strength_log <- log1p(strength_matrix)

# 创建热图
col_fun1 <- colorRamp2(c(0, 2, 4, 6, 8), 
                       c("#FFFFFF", "#C6DBEF", "#6BAED6", "#2171B5", "#08306B"))

col_fun2 <- colorRamp2(c(0, 0.5, 1, 1.5, 2), 
                       c("#FFFFFF", "#C6DBEF", "#6BAED6", "#2171B5", "#08306B"))

ht_length <- Heatmap(
    length_log[1:50, 1:50],
    name = "Length",
    col = col_fun1,
    cluster_rows = TRUE,
    cluster_columns = TRUE,
    column_title = "Projection Length (log)",
    width = unit(8, "cm")
)

ht_strength <- Heatmap(
    strength_log[1:50, 1:50],
    name = "Strength",
    col = col_fun2,
    cluster_rows = TRUE,
    cluster_columns = TRUE,
    column_title = "Projection Strength (log)",
    width = unit(8, "cm")
)

pdf("length_vs_strength.pdf", width = 18, height = 10)
draw(ht_length + ht_strength)
dev.off()
```
